<script>
  (function() {
    function normalizePath(path) {
      if (!path) return '/';

      let normalized = path;

      try {
        normalized = decodeURIComponent(normalized);
      } catch (_) {
        // If the segment is malformed URI encoding, keep the raw value.
      }

      normalized = normalized.replace(/\/{2,}/g, '/');

      if (!normalized.startsWith('/')) {
        normalized = '/' + normalized;
      }

      if (normalized.length > 1 && normalized.endsWith('/')) {
        normalized = normalized.slice(0, -1);
      }

      return normalized || '/';
    }

    function sanitizePathSegment(segment) {
      if (!segment) return '';
      return segment.replace(/^[`'"“”‘’.,;:!?()[\]{}<>]+|[`'"“”‘’.,;:!?()[\]{}<>]+$/g, '');
    }

    function getRequestedPath() {
      const pathname = window.location.pathname || '/';
      return pathname.endsWith('/404.html') ? pathname.slice(0, -9) || '/' : pathname;
    }

    function buildHealingCandidates(path) {
      const candidateSet = new Set();
      const addCandidate = (value) => {
        const normalized = normalizePath(value);
        if (normalized) {
          candidateSet.add(normalized);
        }
      };

      addCandidate(path);

      const parts = normalizePath(path).split('/').filter(Boolean);
      const trimmedParts = [...parts];

      while (trimmedParts.length > 0 && sanitizePathSegment(trimmedParts[trimmedParts.length - 1]) === '') {
        trimmedParts.pop();
        addCandidate('/' + trimmedParts.join('/'));
      }

      if (trimmedParts.length > 0) {
        const lastIndex = trimmedParts.length - 1;
        const sanitizedLastPart = sanitizePathSegment(trimmedParts[lastIndex]);

        if (sanitizedLastPart && sanitizedLastPart !== trimmedParts[lastIndex]) {
          const exactParts = [...trimmedParts];
          exactParts[lastIndex] = sanitizedLastPart;
          addCandidate('/' + exactParts.join('/'));
        }

        const sanitizedAllParts = trimmedParts.map(sanitizePathSegment).filter(Boolean);
        if (sanitizedAllParts.length > 0) {
          addCandidate('/' + sanitizedAllParts.join('/'));
        }
      }

      return Array.from(candidateSet);
    }

    function hideSuggestions() {
      const suggestionsContainer = document.getElementById('searchSuggestions');

      if (suggestionsContainer) {
        suggestionsContainer.style.display = 'none';
      }
    }

    function getSearchTerm() {
      const pathSegments = getRequestedPath().split('/').filter(Boolean);
      let searchTerm = pathSegments[pathSegments.length - 1] || '';

      if (searchTerm.toLowerCase() === '404.html' && pathSegments.length > 1) {
        searchTerm = pathSegments[pathSegments.length - 2];
      }

      try {
        searchTerm = decodeURIComponent(searchTerm);
      } catch (_) {
        // If the segment is malformed URI encoding, keep the raw value.
      }

      return searchTerm.replace(/[-_]+/g, ' ').trim();
    }

    document.addEventListener('DOMContentLoaded', function() {
      const searchTerm = getSearchTerm();
      const suggestionsList = document.getElementById('suggestionsList');
      const requestedPath = getRequestedPath();
      const requestedPathNormalized = normalizePath(requestedPath);
      const worker = new Worker('/js/fuse-search-worker.js');

      worker.onmessage = function (event) {
        const { type, data } = event.data;

        if (type === 'ready') {
          worker.postMessage({ type: 'match-url', data: { variants: buildHealingCandidates(requestedPath) } });
          return;
        }

        if (type === 'error') {
          console.warn('404 suggestions search worker error', data.message);
          hideSuggestions();
          worker.terminate();
          return;
        }

        if (type === 'matched-url') {
          const matchedUrl = data.item && data.item.url ? data.item.url : null;

          if (matchedUrl && normalizePath(matchedUrl) !== requestedPathNormalized) {
            window.location.replace(matchedUrl + window.location.search + window.location.hash);
            return;
          }

          if (!searchTerm || !suggestionsList) {
            hideSuggestions();
            worker.terminate();
            return;
          }

          worker.postMessage({ type: 'search', data: { query: searchTerm, limit: 5 } });
          return;
        }

        if (type === 'results') {
          const { results } = data;

          if (!results || results.length === 0) {
            hideSuggestions();
            worker.terminate();
            return;
          }

          const fragment = document.createDocumentFragment();

          results.forEach((result) => {
            if (!result.item || !result.item.url || !result.item.title) return;

            const li = document.createElement('li');
            const a = document.createElement('a');
            a.href = result.item.url;
            a.textContent = result.item.title;
            li.appendChild(a);
            fragment.appendChild(li);
          });

          if (fragment.childNodes.length === 0) {
            hideSuggestions();
          } else {
            suggestionsList.appendChild(fragment);
          }

          worker.terminate();
        }
      };

      worker.postMessage({
        type: 'init',
        data: {
          options: {
            ignoreLocation: true,
            threshold: 0.3,
            keys: ['title', 'url']
          }
        }
      });
    });
  })();
</script>
