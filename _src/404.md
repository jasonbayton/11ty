<script>
  (function() {
    function hideSuggestions() {
      const suggestionsContainer = document.getElementById('searchSuggestions');

      if (suggestionsContainer) {
        suggestionsContainer.style.display = 'none';
      }
    }

    function getSearchTerm() {
      const pathSegments = window.location.pathname.split('/').filter(Boolean);
      let searchTerm = pathSegments[pathSegments.length - 1] || '';

      if (searchTerm.toLowerCase() === '404.html' && pathSegments.length > 1) {
        searchTerm = pathSegments[pathSegments.length - 2];
      }

      try {
        searchTerm = decodeURIComponent(searchTerm);
      } catch (_) {
        // If the segment is malformed URI encoding, keep the raw value.
      }

      return searchTerm.replace(/[-_]+/g, ' ').trim();
    }

    document.addEventListener('DOMContentLoaded', function() {
      const searchTerm = getSearchTerm();
      const suggestionsList = document.getElementById('suggestionsList');

      if (!searchTerm || !suggestionsList) {
        hideSuggestions();
        return;
      }

      const worker = new Worker('/js/fuse-search-worker.js');

      worker.onmessage = function (event) {
        const { type, data } = event.data;

        if (type === 'ready') {
          worker.postMessage({ type: 'search', data: { query: searchTerm, limit: 5 } });
          return;
        }

        if (type === 'error') {
          console.warn('404 suggestions search worker error', data.message);
          hideSuggestions();
          worker.terminate();
          return;
        }

        if (type === 'results') {
          const { results } = data;

          if (!results || results.length === 0) {
            hideSuggestions();
            worker.terminate();
            return;
          }

          const fragment = document.createDocumentFragment();

          results.forEach((result) => {
            if (!result.item || !result.item.url || !result.item.title) return;

            const li = document.createElement('li');
            const a = document.createElement('a');
            a.href = result.item.url;
            a.textContent = result.item.title;
            li.appendChild(a);
            fragment.appendChild(li);
          });

          if (fragment.childNodes.length === 0) {
            hideSuggestions();
          } else {
            suggestionsList.appendChild(fragment);
          }

          worker.terminate();
        }
      };

      worker.postMessage({
        type: 'init',
        data: {
          options: {
            ignoreLocation: true,
            threshold: 0.3,
            keys: ['title', 'url']
          }
        }
      });
    });
  })();
</script>
